int cantor(const vector<int> &v)
{
	int ans = 0;
	for (int i = 0; i < (int)v.size() - 1; i++)
	{
		int small = 0;
		for (size_t j = i + 1; j < v.size(); j++)
			small += v[j] < v[i];
		ans = (ans + small) * (v.size() - i - 1);
	}
	return ans;
}
